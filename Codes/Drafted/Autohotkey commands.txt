; Create the tray menu
Menu, Tray, Add, Open Notepad, OpenNotepad
Menu, Tray, Add, Exit, ExitScript

Menu, Tray, Default, Open Menu
return 
; Enable double-click action for the tray icon
SysGet, MonitorCount, MonitorCount
TrayDC := 0
SetTimer, TrayDoubleClick, 300

TrayDoubleClick:
    TrayDC += 1
    if (TrayDC = 2)
    {
        TrayDC := 0
        ExitScript:
            ExitApp
    }
    SetTimer, TrayDoubleClick, Off
    SetTimer, TrayDoubleClick, 300
return

OpenNotepad:
    Run, Notepad.exe
return
